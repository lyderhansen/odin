#!/usr/bin/env bash
# tools/tests/check-version-sync.sh — HARD-01 drift guard

set -u

# Portable version extractor — grep + sed, no awk/perl/python dependency.
extract_appconf() {
    local file="$1"
    grep -E '^version\s*=' "$file" 2>/dev/null | head -1 | sed -E 's/^version\s*=\s*//' | tr -d ' '
}

extract_shell_export() {
    local file="$1"
    grep -E '^export ODIN_VERSION=' "$file" 2>/dev/null | head -1 | sed -E 's/^export ODIN_VERSION=//' | tr -d '"' | tr -d "'"
}

extract_ps1_env() {
    local file="$1"
    grep -E "env:ODIN_VERSION\s*=\s*'[^']+'" "$file" 2>/dev/null | head -1 | sed -E "s/.*env:ODIN_VERSION\s*=\s*'([^']+)'.*/\1/"
}

# Resolve paths relative to repo root, not CWD
REPO_ROOT="$(cd "$(dirname "${BASH_SOURCE[0]}")/../.." && pwd)"

ta_odin_version=$(extract_appconf "$REPO_ROOT/TA-ODIN/default/app.conf")
sh_app_version=$(extract_appconf "$REPO_ROOT/ODIN_app_for_splunk/default/app.conf")
odin_sh_version=$(extract_shell_export "$REPO_ROOT/TA-ODIN/bin/odin.sh")
odin_ps1_version=$(extract_ps1_env "$REPO_ROOT/TA-ODIN/bin/odin.ps1")

# Assert all 4 extractions produced non-empty output
for pair in \
    "TA-ODIN/default/app.conf:$ta_odin_version" \
    "ODIN_app_for_splunk/default/app.conf:$sh_app_version" \
    "TA-ODIN/bin/odin.sh:$odin_sh_version" \
    "TA-ODIN/bin/odin.ps1:$odin_ps1_version"; do
    file="${pair%%:*}"
    ver="${pair##*:}"
    if [[ -z "$ver" ]]; then
        echo "[HARD-01 FAIL] Could not extract version from $file"
        exit 1
    fi
done

# Compare all 4 against the TA-ODIN/default/app.conf as canonical
canonical="$ta_odin_version"
drift=0

for pair in \
    "ODIN_app_for_splunk/default/app.conf:$sh_app_version" \

    fi

    echo "[HARD-01 PASS] Version sync: $canonical"
fi

exit $drift

**Explanation of the idiom:**
- `: "${VAR:=default}"` — the leading `:` is bash's no-op builtin. The `${VAR:=default}` form assigns `default` to `VAR` *only if* `VAR` is currently unset or empty. Pre-set env vars pass through unchanged.

**Leave `ODIN_VERSION` alone.** Task 1 already set it to `"1.0.0"` via the direct `export ODIN_VERSION="1.0.0"` pattern. That is correct for HARD-01 (the version should NOT be runtime-overridable — it's the TA-ODIN release string, not a tunable guardrail).

**Shellcheck safety:** The `: "${VAR:=default}"` form is explicitly recommended by shellcheck and will not produce SC2016 or SC2155 findings. The standalone `export VAR1 VAR2` on its own line is also clean.

**Key semantics:**
- `if (-not $env:ODIN_MAX_EVENTS)` — triggers if env var is unset OR empty string (PowerShell truthiness treats both as falsy, matching bash `:=` semantics)
