```powershell
try {
    $services = Get-Service -ErrorAction Stop
    # Fetch start-mode map in ONE Get-CimInstance call (batch — parity with services.sh batch systemctl)
    $startModeMap = @{}
    try {
        $cim = Get-CimInstance -ClassName Win32_Service -ErrorAction Stop
        foreach ($c in $cim) {
            $startModeMap[$c.Name] = $c.StartMode
        }
    } catch {
        # CIM not available — start mode will be "unknown" for every entry (non-fatal)
    }

    foreach ($svc in $services) {
        try {
            $name = Format-OdinValue $svc.Name
            # Map PowerShell Status enum to lowercase parity string
            $status = switch ($svc.Status) {
                'Running'        { 'running' }
                'Stopped'        { 'stopped' }
                'Paused'         { 'paused' }
                'StartPending'   { 'starting' }
                'StopPending'    { 'stopping' }
                'ContinuePending'{ 'resuming' }
                'PausePending'   { 'pausing' }
                default          { 'unknown' }
            }
            # Map Win32_Service StartMode to parity string (enabled|disabled|unknown)
            $rawMode = $startModeMap[$svc.Name]
            $enabled = switch ($rawMode) {
                'Auto'     { 'enabled' }
                'Manual'   { 'enabled' }
                'Disabled' { 'disabled' }
                default    { 'unknown' }
            }
            Invoke-OdinEmit "type=service service_name=$name service_status=$status service_enabled=$enabled"
        } catch {
            # Per-entry fail-soft — one bad service cannot break enumeration.
            # Exception is swallowed; next service continues. Parent orchestrator's
            # module-level timeout and error counter is unaffected.
            continue
        }
    }
} catch {
    # Module-level failure — rethrow so the orchestrator Start-Job wrapper
    # catches it and synthesizes type=odin_error module=services exit_code=1 (WIN-12).
    throw
}
```

```powershell
try {
    # TCP listeners
    $tcp = @()
    try {
        $tcp = Get-NetTCPConnection -State Listen -ErrorAction Stop
    } catch {
        # NetTCPIP module not available — Windows 7 / some locked-down images.
        # Emit nothing and continue to UDP; parity with Linux ss/netstat fallback behavior.
    }

    foreach ($c in $tcp) {
        try {
            $addr = Format-OdinValue $c.LocalAddress
            $port = $c.LocalPort
            $line = "type=port transport=tcp listen_address=$addr listen_port=$port"
            $procPid = $c.OwningProcess
            if ($procPid -and $procPid -gt 0) {
                try {
                    $proc = Get-Process -Id $procPid -ErrorAction Stop
                    $pname = Format-OdinValue $proc.ProcessName
                    $line = "$line process_name=$pname process_pid=$procPid"
                } catch {
                    # Process exited between enumerations — emit the port event without process info (parity with Linux ss behavior on racy PIDs)
                    $line = "$line process_pid=$procPid"
                }
            }
            Invoke-OdinEmit $line
        } catch {
            continue
        }
    }

    # UDP endpoints
    $udp = @()
    try {
        $udp = Get-NetUDPEndpoint -ErrorAction Stop
    } catch {
        # Same defensive fallback as TCP branch
    }

    foreach ($u in $udp) {
        try {
            $addr = Format-OdinValue $u.LocalAddress
            $port = $u.LocalPort
            $line = "type=port transport=udp listen_address=$addr listen_port=$port"
            $procPid = $u.OwningProcess
            if ($procPid -and $procPid -gt 0) {
                try {
                    $proc = Get-Process -Id $procPid -ErrorAction Stop
                    $pname = Format-OdinValue $proc.ProcessName
                    $line = "$line process_name=$pname process_pid=$procPid"
                } catch {
                    $line = "$line process_pid=$procPid"
                }
            }
            Invoke-OdinEmit $line
        } catch {
            continue
        }
    }
} catch {
    throw
}
```

```powershell
# Parity target: TA-ODIN/bin/modules/packages.sh
# Emits: type=package package_name=<n> package_version=<v> [package_arch=<a>] package_manager=registry

# that rewrites every installed MSI's ProductState, can take 5-30 minutes per host,
# and has historically taken down production fleets. Dimension 3 of the Nyquist

# if it finds any match.
#

# Get-ChildItem + Get-ItemProperty. This is the same mechanism Add/Remove Programs
# uses; it is non-invasive and cannot trigger MSI repair.
#
```

**Enumeration block:**
```powershell
$uninstallPaths = @(
    'HKLM:\SOFTWARE\Microsoft\Windows\CurrentVersion\Uninstall',
    'HKLM:\SOFTWARE\Wow6432Node\Microsoft\Windows\CurrentVersion\Uninstall'
)

foreach ($path in $uninstallPaths) {
    try {
        if (-not (Test-Path $path)) { continue }
        $keys = Get-ChildItem -Path $path -ErrorAction Stop
    } catch {
        # Hive path unreadable (unusual permission state under LocalSystem is
        # unexpected, but defensively continue — WIN-11 fail-soft).
        continue
    }

    foreach ($key in $keys) {
        try {
            $props = Get-ItemProperty -Path $key.PSPath -ErrorAction Stop

            # Filter: skip updates/hotfixes and component-only entries.
            # Parity with Add/Remove Programs visible set:
            #   - SystemComponent=1 means "hidden system component" — skip
            #   - ParentKeyName present means "update/patch of another product" — skip
            if ($props.SystemComponent -eq 1) { continue }
            if ($props.PSObject.Properties.Name -contains 'ParentKeyName' -and $props.ParentKeyName) { continue }

            $displayName = $props.DisplayName
            if (-not $displayName) { continue }  # Unnamed entries are installer leftovers — skip

            $name = Format-OdinValue $displayName
            $version = if ($props.DisplayVersion) { Format-OdinValue $props.DisplayVersion } else { 'unknown' }

            $line = "type=package package_name=$name package_version=$version"

            # Optional package_arch — parity with packages.sh dpkg branch which sometimes omits arch.
            # Infer from the hive path: Wow6432Node branch = x86 on 64-bit host; native branch = x64/native.
            $arch = if ($path -like '*Wow6432Node*') { 'x86' } else { 'x64' }
            $line = "$line package_arch=$arch package_manager=registry"

            Invoke-OdinEmit $line
        } catch {
            # Per-entry fail-soft — one corrupt registry value cannot break enumeration.
            continue
        }
    }
}
```

```powershell
    # Mirrors processes.sh's single-ps-call pattern.
    $cimMap = @{}
    try {
        $cim = Get-CimInstance -ClassName Win32_Process -ErrorAction Stop
        foreach ($c in $cim) {
            $cimMap[[int]$c.ProcessId] = $c
        }
    } catch {
        # CIM unavailable — fall back to Get-Process only (no ppid/command/user)
    }

    $procs = Get-Process -ErrorAction Stop

    foreach ($p in $procs) {
        try {
            $procPid = [int]$p.Id
            $name    = Format-OdinValue $p.ProcessName

            # Defaults for optional fields
            $ppid    = 0
            $user    = 'unknown'
            $command = ''

            $cimProc = $cimMap[$procPid]
            if ($cimProc) {
                $ppid    = [int]$cimProc.ParentProcessId
                $command = if ($cimProc.CommandLine) { Format-OdinValue $cimProc.CommandLine } else { '' }

                # Wrap defensively so one ACL-denied process cannot break enumeration.

                        $u = if ($owner.Domain) { "$($owner.Domain)\$($owner.User)" } else { $owner.User }

                } catch {
                    # Keep $user='unknown' — parity with Linux ps fallback

            # to Get-Process are running. Stopped/zombie states are Linux-only semantics.

            # Optional metrics — best effort from Get-Process

            try {
                if ($null -ne $p.CPU) {
                    $cpu = [math]::Round($p.CPU, 2).ToString()
                }
            } catch { }

            try {
                # WorkingSet64 in kilobytes — parity with Linux rss output
                $mem = [int]([long]$p.WorkingSet64 / 1024)
            } catch { }

            try {
                if ($p.StartTime) {
                    $elapsed = [int]((Get-Date) - $p.StartTime).TotalSeconds
                }
            } catch {
                # Access denied on protected processes — omit field

            $line = "type=process process_pid=$procPid process_ppid=$ppid process_user=$user process_state=$state"
            if ($cpu -ne '')     { $line = "$line process_cpu=$cpu" }
            if ($mem -ne '')     { $line = "$line process_mem=$mem" }
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| LocalSystem → Windows Registry hive (`HKLM:\SOFTWARE\...\Uninstall`) | `packages.ps1` reads user-influenceable registry keys. Uninstaller metadata is writable by any installer running as admin — a malicious installer can plant values containing shell metacharacters or terabyte-sized strings. |
| LocalSystem → Win32_Process CIM → GetOwner() | `processes.ps1` invokes `GetOwner()` on every process. ACL-protected processes (csrss, lsass, some AV agents) deny access; error must not break enumeration. |
| Module scriptblock → parent orchestrator via Start-Job output | Modules emit events through `Write-Output`. Start-Job captures stdout and serializes it back to the parent across a process boundary. Malicious module cannot inject into `$script:ODIN_EVENT_COUNT` in the parent because Start-Job is process-isolated (plan 02 T-02-02). |
| `$env:ODIN_TEST_FIXTURE` → stubs.ps1 dot-source | In production, this env var is NEVER set. In fixture mode, `_common.ps1` dot-sources `tools/tests/windows-fixtures/stubs.ps1` — a file under developer control in the repo. |
| Module cmdlet output → Format-OdinValue → Invoke-OdinEmit | Every untrusted-origin string flows through `Format-OdinValue` which strips CR/LF, escapes double-quotes, and wraps in quotes if whitespace or `=` present. |
